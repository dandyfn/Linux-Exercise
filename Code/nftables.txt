
---aktifkan nat pakai nftables---
nft add table ip nat
nft add chain ip nat postrouting { type nat hook postrouting priority 100 \; }
nft add rule ip nat postrouting oif "enp0s3" masquerade

---instalasi nftables---
nano /etc/apt/sources.list
deb http://deb.debian.org/debian trixie main contrib non-free non-free-firmware
deb http://security.debian.org/debian-security trixie-security main contrib non-free non-free-firmware
deb http://deb.debian.org/debian trixie-updates main contrib non-free non-free-firmware
apt clean
apt update
apt install nftables -y
nft --version

---buat nat nftables permanen---
nft list ruleset > /etc/nftables.conf
cat /etc/nftables.conf
systemctl enable nftables
systemctl restart nftables
systemctl status nftables
reboot
nft list ruleset

---konfigurasi resolver di ubuntu (biar bisa ping pakai nama)---
sudo nano /etc/resolv.conf
sudo nano /etc/netplan/00-installer-config.yaml
sudo netplan apply

---konfigurasi resolver di debian (biar bisa ping pakai nama)---
nano /etc/resolv.conf
nameserver 8.8.8.8
nameserver 1.1.1.1
