---virtual host domain based---
cd /etc/apache2
cd /var/www/html
mkdir -p /var/www/domain/dandy
nano /var/www/domain/dandy/index.html
(lalu isikan html)
cd
cd /etc/apache2
cd sites-available
nano dandy-domain.conf
(isikan file ini)
<VirtualHost *:80>
    ServerName dandy.lab
    ServerAlias www.dandy.lab
    DocumentRoot /var/www/domain/dandy

    <Directory /var/www/domain/dandy>
        AllowOverride All
        Require all granted
    </Directory>
</VirtualHost>
a2ensite dandy-domain.conf
a2dissite 000-default.conf
systemctl restart apache2
apachectl -S
(masuk ke klient)
nano /etc/host
(isikan ini)
IP_SERVER   dandy.lab
